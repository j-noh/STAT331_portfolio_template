Writing a function/adding stops

Code 
rescale_01 <- function(vec){
  stopifnot(is.numeric(vec),
            length(vec) > 1)
  
  range <- range(vec, na.rm = TRUE) 
  min <- range[1]
  max <- range[2]
  return((vec-min)/(max-min))
}

rescale_column <- function(vec, variables){
  stopifnot(is.data.frame(vec))
  
  fish |>
